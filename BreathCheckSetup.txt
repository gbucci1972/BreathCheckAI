from fastapi.middleware.cors import CORSMiddleware
from fastapi import FastAPI, File, UploadFile
import numpy as np
import librosa
from pydub import AudioSegment
import tempfile
import os

app = FastAPI()
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # allow all (OK for hackathon)
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# ----------------------------
# Simple AI-like analysis
# ----------------------------
def analyze_audio(y, sr):
    energy = np.mean(y**2)
    zcr = np.mean(librosa.feature.zero_crossing_rate(y))
    spectral_centroid = np.mean(librosa.feature.spectral_centroid(y=y, sr=sr))

    print(f"DEBUG -> Energy: {energy}, ZCR: {zcr}, Spectral: {spectral_centroid}")

    if energy < 0.001:
        return "⚠️ Low breath strength detected"

    if spectral_centroid > 3000:
        return "⚠️ Possible wheeze detected"

    if zcr > 0.15:
        return "⚠️ Irregular breathing pattern"

    return "✅ Normal breathing pattern"


# ----------------------------
# API endpoint
# ----------------------------
@app.post("/analyze")
async def analyze(file: UploadFile = File(...)):
    try:
        # Read uploaded audio
        audio_bytes = await file.read()

        # Save as .webm (browser format)
        with tempfile.NamedTemporaryFile(delete=False, suffix=".webm") as tmp:
            tmp.write(audio_bytes)
            webm_path = tmp.name

        print(f"Saved WEBM: {webm_path}")

        # Convert to WAV using ffmpeg (via pydub)
        wav_path = webm_path.replace(".webm", ".wav")

        audio = AudioSegment.from_file(webm_path, format="webm")
        audio.export(wav_path, format="wav")

        print(f"Converted WAV: {wav_path}")

        # Load audio
        y, sr = librosa.load(wav_path, sr=None)

        print(f"Loaded audio: {len(y)} samples, SR: {sr}")

        # Run analysis
        result = analyze_audio(y, sr)

        # Clean up temp files
        os.remove(webm_path)
        os.remove(wav_path)

        return {"result": result}

    except Exception as e:
        print("ERROR:", str(e))
        return {"result": "❌ Error processing audio"}

<script>
let mediaRecorder;
let audioChunks = [];

async function startRecording() {
  document.getElementById("status").innerText = "Recording...";

  const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
  mediaRecorder = new MediaRecorder(stream);

  audioChunks = [];

  mediaRecorder.ondataavailable = event => {
    console.log("Chunk received");
    audioChunks.push(event.data);
  };

  mediaRecorder.start();
}

async function stopRecording() {
  if (!mediaRecorder) {
    alert("Recording not started!");
    return;
  }

  document.getElementById("status").innerText = "Processing...";

  mediaRecorder.stop();

  mediaRecorder.onstop = async () => {
    const audioBlob = new Blob(audioChunks, { type: 'audio/webm' });

    const formData = new FormData();
    formData.append("file", audioBlob);

    try {
      const response = await fetch("http://127.0.0.1:8000/analyze", {
        method: "POST",
        body: formData
      });

      const data = await response.json();
      document.getElementById("result").innerText = data.result;

      document.getElementById("status").innerText = "Done";
    } catch (error) {
      console.error(error);
      document.getElementById("status").innerText = "Error";
    }
  };
}
</script>

</body>
</html>

🧪 ✅ SIMPLE TEST PROTOCOL (what YOU should do)

You want a structured 3-step test (like a real screening tool)

🎤 STEP 1 — Breathing test (10 seconds)
What to do:
Hold phone ~20 cm from mouth
Quiet room
Do:

👉 3 deep breaths:

INHALE (3 sec) → EXHALE (4 sec)

Repeat 3 times (~10 seconds total)

What the app is “looking for”:
airflow consistency
noise (wheezing-like sounds)
breath interruptions
Expected result:
Normal → low energy variation
No spikes → “green / normal”
💨 STEP 2 — Cough test (5–8 seconds)
What to do:

👉 Do 3 strong coughs, like:

COUGH — pause — COUGH — pause — COUGH
Important:
Not fake throat clearing ❌
Real chest cough ✅
What the app detects:
sharp spikes in amplitude
spectral changes (cough signature)
Expected:
Strong spikes → classified as “cough detected”

"Today I feel well and my breathing is normal"

voice stability

pauses / strain
